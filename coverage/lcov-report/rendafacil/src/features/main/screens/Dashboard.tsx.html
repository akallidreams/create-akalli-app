import {
  VStack,
  Input,
  SearchIcon,
  HStack,
  Text,
  Button,
  ScrollView,
  Icon,
} from 'native-base'
import { useState, useEffect } from 'react'
// import { Filters } from "./Filters";
import { TransactionsHistory, B3Connect, MainHeader } from '../components'
import { b3Bridge } from '../services'
import { actions } from '../redux'
import { RootStateOrAny, useSelector } from 'react-redux'
import {
  GestureResponderEvent,
  NativeSyntheticEvent,
  TextInputChangeEventData,
} from 'react-native'
import { DateTimePickerAndroid } from '@react-native-community/datetimepicker'
import { utils } from '../helpers'
import { useAppDispatch } from '@config/store'
import { Path } from 'react-native-svg'
 
export function Main() {
  const [loading, setLoading] = useState<boolean>(true)
  const dispatch = useAppDispatch()
  // FIXME: It has 2 layers of history in the store
  const { history } = useSelector((state: RootStateOrAny) => {
    return state
  })
  const [transactions, setTransactions] = useState(history)
  const [searchTerm, setSearchTerm] = useState<string>('')
  const [b3Connection, setB3Connection] = useState(false)
  const [dateSelected, setDateSelected] = useState<false | string>(false)
 
  useEffect(() => {
    // clearStore()
    setLoading(true)
    // handleFilter(7)
    setB3Connection(true)
    handleB3(new Date(utils.yesterday))
  }, [])
 
  useEffect(() => {
    setLoading(true)
    setTransactions(history)
    utils.last1sec(() => {
      setLoading(false)
    })
  }, [history])
 
  const handleB3 = (date?: Date) => {
    b3Bridge.getTransactions(date).then(res => {
      // TODO: Add this on redux as the history, 2. show on the page, 3 manage filters and decide if keep them
      dispatch(actions.filterHistory(res.data))
      setLoading(false)
    })
  }
 
  // const handleFilter = (selectedFilter: number) => {
  //   setSearchTerm('')
  //   setLoading(true)
  //   dispatch(actions.filterHistory(selectedFilter)).then(() => {
  //     setTimeout(() => {
  //       setLoading(false)
  //     }, 1000)
  //   })
  // }
 
  const handleSearch = (e: NativeSyntheticEvent<TextInputChangeEventData>) => {
    // FIXME: fix type here
    const { text } = e.nativeEvent
    setSearchTerm(text)
    if (text.length === 0) {
      handleB3()
      return
    }
    setLoading(true)
    utils.last1sec(() => {
      dispatch(actions.searchHistory(text)).then(() => {
        setLoading(false)
      })
    })
  }
 
  const handleDateChange = () => {
    // FIXME: on cancel fix here
    DateTimePickerAndroid.open({
      value: new Date(),
      onChange: ({ nativeEvent }: GestureResponderEvent) => {
        if (nativeEvent.timestamp && nativeEvent.timestamp > +new Date()) {
          setDateSelected(utils.yesterdayFormatedBR)
          handleB3()
        } else {
          const date = new Date(nativeEvent.timestamp as number)
          handleB3(date)
          setDateSelected(date.toLocaleDateString('pt-BR'))
        }
      },
      mode: 'date',
      is24Hour: true,
    })
  }
 
  return (
    <>
      <MainHeader />
      <VStack space="10px" p="5%" bg="white" w="full" mb="60px">
        {!b3Connection ? (
          <B3Connect />
        ) : (
          <ScrollView w="full" h="80%" scrollEnabled borderWidth="3px">
            <HStack alignItems="center">
              <Input
                variant="search"
                value={searchTerm}
                onChange={handleSearch}
                mr="14px"
                InputLeftElement={
                  <SearchIcon
                    name="search"
                    color="grey"
                    size="20px"
                    mx="10px"
                  />
                }
              />
              <Button
                borderWidth="1px"
                borderColor="grey"
                borderRadius="6px"
                h="40px"
                p="3px"
                px="10px"
                bg="extra.greyExtraLight"
                onPress={handleDateChange}>
                <HStack alignItems="center" space="6px">
                  <Text variant="primary" color="grey">
                    Data
                  </Text>
                  <OpenDateIcon />
                </HStack>
              </Button>
            </HStack>
            {/* <Filters selectFilter={handleFilter} /> */}
            <TransactionsHistory
              history={transactions}
              loading={loading}
              dateSelected={dateSelected}
            />
          </ScrollView>
        )}
      </VStack>
    </>
  )
}
